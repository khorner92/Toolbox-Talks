<script>

const STORAGE_KEY = 'tbt_v1';

let entries = [];

let adminUnlocked = false;


// Apply config immediately

document.getElementById('hdr-company').textContent  = CONFIG.company;

document.getElementById('di-title').textContent     = CONFIG.talkTitle;

document.getElementById('di-presenter').textContent = CONFIG.presenter;

document.getElementById('ack-title').textContent    = CONFIG.talkTitle;

document.getElementById('di-date').textContent      = new Date().toLocaleDateString('en-US',

  { weekday:'long', year:'numeric', month:'long', day:'numeric' });


// ── CANVAS ──

const canvas = document.getElementById('sigCanvas');

const ctx = canvas.getContext('2d');

let drawing = false, hasSig = false;


function setupCtx() { ctx.strokeStyle='#F5C400'; ctx.lineWidth=2.2; ctx.lineCap='round'; ctx.lineJoin='round'; }

function resizeCanvas() {

  const w = Math.min(660, canvas.parentElement.clientWidth - 2);

  const snap = hasSig ? canvas.toDataURL() : null;

  canvas.width = w; canvas.height = 120; setupCtx();

  if (snap) { const img = new Image(); img.onload = () => ctx.drawImage(img,0,0,w,120); img.src = snap; }

}

setupCtx(); resizeCanvas();

window.addEventListener('resize', resizeCanvas);


function getPos(e) {

  const r = canvas.getBoundingClientRect(), s = e.touches ? e.touches[0] : e;

  return { x:(s.clientX-r.left)*(canvas.width/r.width), y:(s.clientY-r.top)*(canvas.height/r.height) };

}

canvas.addEventListener('mousedown',  e => { drawing=true; const p=getPos(e); ctx.beginPath(); ctx.moveTo(p.x,p.y); });

canvas.addEventListener('mousemove',  e => { if(!drawing)return; const p=getPos(e); ctx.lineTo(p.x,p.y); ctx.stroke(); hasSig=true; });

canvas.addEventListener('mouseup',    () => drawing=false);

canvas.addEventListener('mouseleave', () => drawing=false);

canvas.addEventListener('touchstart', e => { e.preventDefault(); drawing=true; const p=getPos(e); ctx.beginPath(); ctx.moveTo(p.x,p.y); }, {passive:false});

canvas.addEventListener('touchmove',  e => { e.preventDefault(); if(!drawing)return; const p=getPos(e); ctx.lineTo(p.x,p.y); ctx.stroke(); hasSig=true; }, {passive:false});

canvas.addEventListener('touchend',   () => drawing=false);

function clearSig() { ctx.clearRect(0,0,canvas.width,canvas.height); hasSig=false; }


// ── STORAGE ──

async function saveEntries() { try { await window.storage.set(STORAGE_KEY+'_entries', JSON.stringify(entries)); } catch(e){} }

async function loadEntries()  { try { const r=await window.storage.get(STORAGE_KEY+'_entries'); if(r) entries=JSON.parse(r.value); } catch(e){ entries=[]; } }


// ── SUBMIT ──

function submitEntry() {

  const first=document.getElementById('f-first').value.trim();

  const last =document.getElementById('f-last').value.trim();

  const ack  =document.getElementById('ack-check').checked;

  const err  =document.getElementById('form-error');

  if (!first||!last||!hasSig||!ack) { err.style.display='block'; return; }

  err.style.display='none';

  entries.push({ id:Date.now(), first, last, sig:canvas.toDataURL('image/png'),

    time:new Date().toLocaleString('en-US',{month:'short',day:'numeric',year:'numeric',hour:'numeric',minute:'2-digit'}) });

  saveEntries(); renderRoster('roster-container');

  document.getElementById('f-first').value='';

  document.getElementById('f-last').value='';

  document.getElementById('ack-check').checked=false;

  clearSig();

  const f=document.getElementById('success-flash'); f.style.display='block';

  setTimeout(()=>f.style.display='none', 3000);

}


// ── ROSTER ──

function renderRoster(id) {

  const el=document.getElementById(id);

  const cnt=document.getElementById('roster-count');

  if(cnt) cnt.textContent=entries.length+' SIGNED';

  if(!entries.length){ el.innerHTML='<div class="roster-empty">No signatures yet — be the first to sign in.</div>'; return; }

  el.innerHTML=`<table class="roster-table">

    <thead><tr><th>#</th><th>Name</th><th>Signature</th><th>Timestamp</th></tr></thead>

    <tbody>${entries.map((e,i)=>`<tr>

      <td class="entry-num">#${i+1}</td>

      <td><strong>${e.first} ${e.last}</strong></td>

      <td><img class="sig-thumb" src="${e.sig}" alt="sig"/></td>

      <td style="font-family:'IBM Plex Mono',monospace;font-size:12px;color:var(--muted);">${e.time}</td>

    </tr>`).join('')}</tbody>

  </table>`;

}


// ── TABS ──

function switchTab(tab) {

  ['signin','admin'].forEach(t=>{

    document.getElementById('view-'+t).classList.toggle('active', t===tab);

    document.getElementById('tab-'+t).classList.toggle('active', t===tab);

  });

  if(tab==='admin') refreshAdmin();

}


// ── ADMIN ──

function checkPin() {

  if(document.getElementById('pin-input').value===CONFIG.adminPin) {

    document.getElementById('pin-gate').style.display='none';

    document.getElementById('admin-panel').style.display='block';

    adminUnlocked=true; refreshAdmin();

  } else { document.getElementById('pin-error').style.display='block'; }

}

function refreshAdmin() {

  if(!adminUnlocked) return;

  document.getElementById('admin-talk-title').textContent=CONFIG.talkTitle;

  document.getElementById('admin-count').textContent=entries.length;

  document.getElementById('admin-date').textContent=document.getElementById('di-date').textContent;

  document.getElementById('admin-presenter').textContent=CONFIG.presenter;

  renderRoster('admin-roster');

}

function exportCSV() {

  let csv=`"Talk Title","${CONFIG.talkTitle}"\n"Company","${CONFIG.company}"\n"Presenter","${CONFIG.presenter}"\n"Date","${document.getElementById('di-date').textContent}"\n\n"#","First Name","Last Name","Timestamp"\n`;

  entries.forEach((e,i)=>csv+=`"${i+1}","${e.first}","${e.last}","${e.time}"\n`);

  const a=document.createElement('a');

  a.href=URL.createObjectURL(new Blob([csv],{type:'text/csv'}));

  a.download=`toolbox-talk-${CONFIG.talkTitle.replace(/\s+/g,'-').toLowerCase()}.csv`;

  a.click();

}

function clearEntries() {

  if(!confirm('Clear ALL sign-in entries? This cannot be undone.')) return;

  entries=[]; saveEntries(); refreshAdmin(); renderRoster('roster-container');

}


// ── INIT ──

(async()=>{ await loadEntries(); renderRoster('roster-container'); })();
